  const processNumber = () => {
    const num = parseInt(document.getElementById("inputNum").value);     const resultDiv = document.getElementById("result");     const checkPrime = document.getElementById("chkPrime").checked;     const checkFactorial = document.getElementById("chkFactorial").checked;     const checkFibonacci = document.getElementById("chkFibonacci").checked;     if (isNaN(num) || num < 0) {
      resultDiv.innerHTML = "Please enter a valid non-negative number.";       resultDiv.style.display = "block";       return;     }
    let output = "";     if (checkPrime) {       let isPrime = true;       if (num < 2) {         isPrime = false;
      } else {
        for (let i = 2; i <= Math.sqrt(num); i++) {           if (num % i === 0) {             isPrime = false;
            break;           }      }    }
      output += `<strong>${num}</strong> is ${isPrime ? "a Prime" : "NOT a Prime"} number.<br>`;
    }
    if (checkFactorial) {       let factorial = 1;
      for (let i = 1; i <= num; i++) {
        factorial *= i;
      }
      output += `Factorial of <strong>${num}</strong> is <strong>${factorial}</strong>.<br>`;
    }
    if (checkFibonacci) {       let fibSeries = "";       let a = 0, b = 1;
      for (let i = 0; i < num; i++) {         fibSeries += a + " ";         let temp = a + b;
a	= b;
b	= temp;
      }
      output += `First <strong>${num}</strong> numbers of Fibonacci series:<br><strong>${fibSeries}</strong>`;
    }
    if (!checkPrime && !checkFactorial && !checkFibonacci) {       output = "Please select at least one operation.";
    }
    resultDiv.innerHTML = output;     resultDiv.style.display = "block";
 };
